import { Component, Input, OnInit } from '@angular/core';
import { BaseComponent } from '../shared/base.component';
import { COLOR_WHITE, COLOR_BLUE } from '../shared/constants';

@Component({
  selector: 'fj-button',
  templateUrl: './button.component.html',
  styleUrls: ['./button.component.scss'],
  host: {
    'class': 'fj-button',
  }
})
export class ButtonComponent extends BaseComponent implements OnInit {
  /**
   * @description Freyja Button Component using Freyja Design specifics
   */


  @Input() color;
  @Input() ghost = false;
  @Input() type = 'button';
  @Input() shape: 'radius' | 'square' | 'rounded' = 'radius';
  @Input() size: 'micro' | 'small' | 'medium' | 'large' = 'medium';

  /**
   * Inputs Specific to Icon Component
   */
  @Input() icon;
  @Input() iconMargin;
  @Input() iconFill = COLOR_WHITE;
  @Input() iconPosition: 'after' | 'before' = 'before';

  // TODO: JSDOCS
  // TODO: add generic generateClass methods

  constructor() {
    super();
  }

  ngOnInit(): void {
    if (this.ghost) {
      this.iconFill = COLOR_BLUE;
    }
  }
  /**
   * @description Add css classes based on inputs like size, shape etc
   */
  public generateClassList() {
    const baseCls = 'fj-button--';
    const classes = {
      [baseCls + this.color]: this.color,
      [baseCls + this.size]: this.size,
      [baseCls + this.shape]: this.shape,
      [`${baseCls}icon-before`]: this.iconPosition === 'before',
      [`${baseCls}ghost`]: this.ghost
    };
    return classes;
  }
}
