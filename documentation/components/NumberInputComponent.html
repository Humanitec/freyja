import { Component, ViewEncapsulation, Input, Output, EventEmitter } from '@angular/core';
import { BaseComponent} from '../shared/base.component';

let nextId = 0;

@Component({
  selector: 'fj-number-input',
  templateUrl: './number-input.component.html',
  styleUrls: ['./number-input.component.scss'],
  encapsulation: ViewEncapsulation.None,
})
export class NumberInputComponent extends BaseComponent {


  private uniqueInputId = `fj-number-input-${++nextId}`;
  private isErrored =  false;
  private inputValue = '';
  @Input() id: string = this.uniqueInputId;

  get inputId(): string { return `${this.id || this.inputId}-input`; }

  @Input() placeholder = '';
  @Input() label = '';
  @Input() value = this.inputValue;
  @Input() size: 'micro' | 'small' | 'med' = 'med';
  @Input() errored = false;

  @Output() change: EventEmitter<any> = new EventEmitter<any>();

  constructor() {
    super();
   }

  onValueChange(event: Event) {
    event.stopPropagation();
    this.inputValue = (<HTMLInputElement>event.target).value;
    this.change.emit(this.inputValue);
  }

  public generateClassList() {
    const classes = {
      ['fj-number-input--' + this.size]: this.size,
      ['fj-number-input--error']: this.errored,
    };
    return classes;
  }

}
